         if(isdigit(s[i])){if (num == 0 && s[i] - '0' == 0)
		cout << "Enter Valid IP Address\n";exit(0);}
		num *= 10;
		num+=(s[i]-'0');}
	elseif (s[i] == '0'){
		if(i==0||(i-1)>=0&& s[i-1]=='.'){
		cout<<"Enter Valid IP Address\n";}
exit(0);}
            }
    a.push_back(num);

        cout << "Enter valid IP address\n";

    }
    for (int i = 0; i < 4; i++) {
        if(a[i] > 255) {
            cout << "Enter valid IP address\n";
            exit(0);
        }
    }
    cout << "Valid IP address\n";
    if(a[0] <= 127) {
        cout << "The IP address is of class A\n";

        cout << "The IP address is of class B\n";

}
